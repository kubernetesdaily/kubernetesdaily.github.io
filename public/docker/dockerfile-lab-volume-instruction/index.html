Dockerfile Lab - Volume instruction

cat dockerfile.vol

FROM alpine
RUN mkdir /myvol
RUN echo "hello world" > /myvol/greeting
VOLUME /myvol
CMD ["sh"]

docker build -t sangam14/vol-dockerfile  -f dockerfile.vol .
